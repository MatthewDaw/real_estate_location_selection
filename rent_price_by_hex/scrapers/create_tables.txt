CREATE TABLE landwatch_properties (
    id SERIAL PRIMARY KEY,

    -- Identification & Metadata
    name TEXT,
    title TEXT,
    url TEXT,
    listing_date DATE,
    price NUMERIC,

    -- Location Information
    address1 TEXT,
    address2 TEXT,
    city TEXT,
    zip TEXT,
    county TEXT,
    state TEXT,
    latitude NUMERIC,
    longitude NUMERIC,
    city_latitude NUMERIC,
    city_longitude NUMERIC,

    -- Lot Details
    lot_size NUMERIC,
    lot_size_units TEXT,
    acres NUMERIC,
    lot_type TEXT[],
    lot_description TEXT[],
    road_frontage_desc TEXT[],
    is_irrigated BOOLEAN,

    -- Property Features
    beds INTEGER,
    baths INTEGER,
    homesqft INTEGER,
    is_residence BOOLEAN,
    property_type TEXT,
    property_types TEXT[],

    -- Additional Attributes
    amenities TEXT[],
    mortgage_options TEXT[],
    activities TEXT[],
    geography TEXT[],

    -- Description & Summary
    description TEXT[],
    executive_summary TEXT,

    -- Tier Flags
    is_diamond BOOLEAN,
    is_gold BOOLEAN,
    is_platinum BOOLEAN,
    is_showcase BOOLEAN,

    -- Computed Fields
    cost_per_acre NUMERIC,
    distance_to_city_miles NUMERIC,
    cost_per_homesqft NUMERIC
);


CREATE TABLE landwatch_urls (
    url TEXT NOT NULL,
    created_on DATE DEFAULT CURRENT_DATE,
    scraped_at DATE DEFAULT NULL,
    state CHAR(2) DEFAULT NULL
);



CREATE TABLE zillow_urls (
    id SERIAL PRIMARY KEY, -- auto-incrementing id
    url TEXT NOT NULL,
    type TEXT NOT NULL,
    scraped_at DATE DEFAULT NULL,
    created_on DATE DEFAULT CURRENT_DATE,
    UNIQUE (url, type) -- ensure upsert works on both url and type
);

    living_area_units TEXT,
    living_area NUMERIC,
    street_address TEXT,
    city TEXT,
    state TEXT,
    zipcode TEXT,
    price NUMERIC,
    currency TEXT,
    home_type TEXT,
    is_preforeclosure_auction BOOLEAN,
    address JSONB,
    bedrooms INTEGER,
    bathrooms INTEGER,
    year_built INTEGER,
    living_area_units_short TEXT,
    country TEXT,
    monthly_hoa_fee NUMERIC,
    zestimate NUMERIC,
    new_construction_type TEXT,
    zestimate_low_percent NUMERIC,
